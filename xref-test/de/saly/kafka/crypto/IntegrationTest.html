78          
79          EmbeddedZookeeper zkServer = new EmbeddedZookeeper();
80          String zkConnect = ZKHOST + ":" + zkServer.port();
81          ZkClient zkClient = new ZkClient(zkConnect, 30000, 30000, ZKStringSerializer$.MODULE$);
82          ZkUtils zkUtils = ZkUtils.apply(zkClient, false);
83          
84          // setup Broker
85          Properties brokerProps = new Properties();
86          brokerProps.setProperty("zookeeper.connect", zkConnect);
87          brokerProps.setProperty("broker.id", "0");
88          final Path tmp = Files.createTempDirectory("kafka-");
89          
90          Runtime.getRuntime().addShutdownHook(new Thread() {
91              public void run() {
92                try {
93                    Files.walkFileTree(tmp, new SimpleFileVisitor<Path>() {
94                        @Override
95                        public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
96                                Files.delete(file);
97                                return FileVisitResult.CONTINUE;
98                        }
99  
100                       @Override
101                       public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
102                               Files.delete(dir);
103                               return FileVisitResult.CONTINUE;
104                       }
105 
106               });
107               System.out.println("Temprary data under "+tmp.toAbsolutePath()+" deleted");
108             } catch (IOException e) {
109                 e.printStackTrace();
110             }
111             }
112           });
113         
114         
115         brokerProps.setProperty("log.dirs", tmp.toAbsolutePath().toString());
116         brokerProps.setProperty("listeners", "PLAINTEXT://" + BROKERHOST +":" + BROKERPORT);
117         KafkaConfig config = new KafkaConfig(brokerProps);
118         Time mock = new MockTime();
119         KafkaServer kafkaServer = TestUtils.createServer(config, mock);
120 
121         try {
122             // create topic
123             AdminUtils.createTopic(zkUtils, TOPIC, 1, 1, new Properties(), RackAwareMode.Disabled$.MODULE$);
124             
125             Properties producerProps = new Properties();
126             producerProps.setProperty("bootstrap.servers", BROKERHOST + ":" + BROKERPORT);
127             producerProps.put("enable.auto.commit", "true");
128             producerProps.put("key.serializer", "org.apache.kafka.common.serialization.StringSerializer");
129             producerProps.put("value.serializer", "de.saly.kafka.crypto.EncryptingSerializer");
130             producerProps.put("crypto.wrapped_serializer", "org.apache.kafka.common.serialization.StringSerializer");
131             producerProps.put("crypto.rsa.publickey.filepath", pubKey.getAbsolutePath());
132 
133             try(Producer<String, String> producer = new KafkaProducer<String, String>(producerProps)) {
134                 for(int i = 0; i < 1000; i++) {
135                     producer.send(new ProducerRecord<String, String>(TOPIC, Integer.toString(i), "test:"+Integer.toString(i)));
136                 }
137             }
138             
139             Properties consumerProps = new Properties();
140             consumerProps.setProperty("bootstrap.servers", BROKERHOST + ":" + BROKERPORT);
141             consumerProps.put("group.id", "test");
142             consumerProps.put("key.deserializer", "org.apache.kafka.common.serialization.StringDeserializer");
143             consumerProps.put("value.deserializer", DecryptingDeserializer.class);
144             consumerProps.put("crypto.wrapped_deserializer", "org.apache.kafka.common.serialization.StringDeserializer");
145             consumerProps.put("crypto.rsa.privatekey.filepath", privKey.getAbsolutePath());
146             consumerProps.put("auto.offset.reset", "earliest");
147             
148             try(KafkaConsumer<String, String> consumer = new KafkaConsumer<String, String>(consumerProps)) {
149                 consumer.subscribe(Arrays.asList(TOPIC));
150                 
151                 int i =0;
152                 ConsumerRecords<String, String> records = consumer.poll(10000);
153                 
154                 Assert.assertEquals(1000, records.count());
155                 
156                 for (ConsumerRecord<String, String> record : records) {
157                     assertEquals(String.valueOf(i), record.key());
158                     assertEquals("test:"+Integer.toString(i++), record.value());
159                 }
160             }
161         } finally {
162             kafkaServer.shutdown();
163             kafkaServer.awaitShutdown();
164             zkServer.shutdown();
165         }
